  intent: AuthorChapterIntent,
  lang: "zh" | "en",

  const L = lang === "en"
    ? { title: "## Chapter Intent", core: "Core Narrative", takeaway: "Reader Takeaway",
        moment: "Key Moment", beats: "Required Beats", avoid: "Forbidden Moves",
        hooks: "Pending Hooks", scenes: "Scene Plans", chars: "Character States" }

        hooks: "待推进伏笔", scenes: "场景规划", chars: "角色状态" };

  const lines = [`${L.title} — 第 ${intent.chapterNumber} 章`, ""];

// packages/studio/src/api/server.ts
// 搜索 "event-chain" 定位现有端点

// 改造前（假数据）：
// app.post("/api/v1/books/:id/event-chain/extract", async (c) => {
//   return c.json({ nodes: MOCK_EVENT_NODES });  ← 假数据
// });

// 改造后（真实 Agent 调用）：
app.post("/api/v1/books/:id/event-chain/extract", async (c) => {
  const id = c.req.param("id");
  await assertBookExists(state, id);
  const bookDir = state.bookDir(id);
  
  // 1. 收集所有已写章节的正文
  const { readFile, readdir } = await import("node:fs/promises");
  const { join } = await import("node:path");
  const storyDir = join(bookDir, "story");
  const chapterFiles = (await readdir(storyDir))
    .filter((f) => /^chapter-\d{4}\.md$/.test(f))
    .sort();
  
  const chapterContents: string[] = [];
  for (const file of chapterFiles) {
    const content = await readFile(join(storyDir, file), "utf-8");
    chapterContents.push(content);
  }
  
  // 2. 读取角色列表
  const characterDir = join(bookDir, "story", "characters");
  let characters: string[] = [];
  try {
    const charFiles = await readdir(characterDir);
    characters = charFiles
      .filter((f) => f.endsWith(".json"))
      .map((f) => f.replace(/\.json$/, ""));
  } catch { /* 无角色目录 */ }
  
  // 3. 实例化并调用 EventChainExtractor
  const { EventChainExtractor } = await import("@actalk/inkos-core");
  const extractor = new EventChainExtractor({
    sourceFiles: chapterFiles.map((f) => join(storyDir, f)),
    chapterContents,
    characters,
    language: "zh",
  });
  const nodes = await extractor.execute();
  
  // 4. 持久化到 artifact
  const artifactPath = join(bookDir, "story", "state", "event_chain.json");
  await writeFile(artifactPath, JSON.stringify({ nodes, updatedAt: new Date().toISOString() }, null, 2), "utf-8");
  
  return c.json({ nodes });
});
```

 * Validate that a written chapter satisfies its declared openingFrame / closingFrame.
 * Called post-write by the Runner to surface endpoint-lock violations.
 */
export function validateEndpointLock(
  chapterContent: string,
  chapterIntent: AuthorChapterIntent,
  chapterNumber: number,
  lang: "zh" | "en",
): EndpointValidation {
  const checks: EndpointCheck[] = [];
  
  // 1. Opening frame check
  if (chapterIntent.openingFrame) {
    const openingWords = chapterContent.slice(0, 200);
    const hasOpening = fuzzyContains(openingWords, chapterIntent.openingFrame);
    checks.push({
      name: lang === "en" ? "Opening Frame" : "开篇框架",
      passed: hasOpening,
      detail: hasOpening
        ? (lang === "en" ? "Opening matches declared frame" : "开篇与声明框架一致")
        : (lang === "en"
          ? `Expected opening to contain: "${chapterIntent.openingFrame}"`
          : `预期开篇应包含："${chapterIntent.openingFrame}"`),
    });
  }
  
  // 2. Closing frame check
  if (chapterIntent.closingFrame) {
    const closingWords = chapterContent.slice(-500);
    const hasClosing = fuzzyContains(closingWords, chapterIntent.closingFrame);
    checks.push({
      name: lang === "en" ? "Closing Frame" : "收尾框架",
      passed: hasClosing,
      detail: hasClosing
        ? (lang === "en" ? "Closing matches declared frame" : "收尾与声明框架一致")
        : (lang === "en"
          ? `Expected closing to contain: "${chapterIntent.closingFrame}"`
          : `预期收尾应包含："${chapterIntent.closingFrame}"`),
    });
  }
  
  // 3. Required beats check
  if (chapterIntent.requiredBeats?.length) {
    for (const beat of chapterIntent.requiredBeats) {
      const found = fuzzyContains(chapterContent, beat);
      checks.push({
        name: lang === "en" ? `Required Beat: ${beat}` : `必达事件：${beat}`,
        passed: found,
        detail: found
          ? (lang === "en" ? "Beat satisfied" : "事件已达成")
          : (lang === "en" ? "Beat NOT found in chapter" : "章节中未发现此事件"),
      });
    }
  }
  
  // 4. Forbidden moves check
  if (chapterIntent.forbiddenMoves?.length) {
    for (const move of chapterIntent.forbiddenMoves) {
      const found = fuzzyContains(chapterContent, move);
      checks.push({
        name: lang === "en" ? `Forbidden: ${move}` : `禁用动作：${move}`,
        passed: !found,
        detail: found
          ? (lang === "en" ? "Forbidden move DETECTED in chapter!" : "章节中发现禁用动作！")
          : (lang === "en" ? "Forbidden move absent" : "未发现禁用动作"),
      });
    }
  }
  
  return {
    passed: checks.every((c) => c.passed),
    checks,
  };
}

/** Simple substring/fuzzy match — can be upgraded to embedding similarity later. */
function fuzzyContains(text: string, pattern: string): boolean {
  // Normalize: lowercase, collapse whitespace
  const norm = (s: string) => s.toLowerCase().replace(/\s+/g, " ").trim();
  const normText = norm(text);
  const normPattern = norm(pattern);
  // 1. Substring match
  if (normText.includes(normPattern)) return true;
  // 2. 70%+ token overlap
  const patternTokens = new Set(normPattern.split(/\s+/).filter((t) => t.length > 1));
  const textTokens = new Set(normText.split(/\s+/).filter((t) => t.length > 1));
  const overlap = [...patternTokens].filter((t) => textTokens.has(t)).length;
  return overlap / patternTokens.size >= 0.7;
}
```

  const max = Math.max(...data, 1);

  return (
