```typescript
// use-api.ts — useApi hook 内的 refetch 函数
const refetch = useCallback(async () => {
  // ... 现有逻辑 ...
  const url = buildApiUrl(path);
  if (!url) { /* ... */ return; }

  // P1-8: 接入 GET 请求去重
  const method = (init?.method ?? "GET").toUpperCase();
  let dedupSignal: AbortSignal | undefined;
  if (method === "GET") {
    dedupSignal = dedupGetRequest(url);
  }

  try {
    const json = await fetchJson<T>(url, {}, {
      signal: dedupSignal ?? controller.signal,
    });
    // ... 现有状态更新逻辑 ...
  } catch (e) {
```
